The menu endpoints provide access to daily cafeteria menus and a comprehensive search across all available meal data.

## Get menu by date

Retrieve the complete menu for a specific date.

```http
GET /api/menu/date/:date
```

### Path parameters

<ParamField path="date" type="string" required>
  Date in ISO format `YYYY-MM-DD` (e.g., `2025-11-03`)
</ParamField>

### Response

<ResponseField name="found" type="boolean">
  Whether menu data exists for the requested date
</ResponseField>

<ResponseField name="day" type="DayMenu" optional>
  Complete menu data for the date (only present if `found` is `true`)
  
  <Expandable title="DayMenu properties">
    <ResponseField name="ymk" type="number">
      Week number within the month (1, 2, 3...)
    </ResponseField>
    
    <ResponseField name="date" type="string">
      ISO date format `YYYY-MM-DD`
    </ResponseField>
    
    <ResponseField name="dayName" type="string">
      Turkish day name (e.g., "Pazartesi", "Salı")
    </ResponseField>
    
    <ResponseField name="hasData" type="boolean">
      Whether meal data is available for this day
    </ResponseField>
    
    <ResponseField name="meals" type="Meal[]">
      Array of meals (empty if `hasData` is `false`)
    </ResponseField>
    
    <ResponseField name="totalCalories" type="number">
      Total calories for all meals on this day
    </ResponseField>
  </Expandable>
</ResponseField>

### Meal object

<ResponseField name="id" type="string">
  Unique meal identifier (e.g., "157")
</ResponseField>

<ResponseField name="name" type="string">
  Meal name (e.g., "Ekşili Köfte")
</ResponseField>

<ResponseField name="calories" type="number">
  Calorie count for the meal
</ResponseField>

<ResponseField name="category" type="MealCategory">
  Meal category: `ana_yemek` (main dish), `yan_yemek` (side), `corba` (soup), `yan_urun` (accompaniment), `tatli` (dessert), or `icecek` (beverage)
</ResponseField>

## Search meals

Search and retrieve all unique meals across all available menu data.

```http
GET /api/menu/search
```

### Response

<ResponseField name="meals" type="MealSearchResult[]">
  Array of unique meals with their appearance dates
  
  <Expandable title="MealSearchResult properties">
    <ResponseField name="name" type="string">
      Meal name (e.g., "Ekşili Köfte")
    </ResponseField>
    
    <ResponseField name="calories" type="number">
      Calorie count for the meal
    </ResponseField>
    
    <ResponseField name="category" type="string">
      Meal category (ana_yemek, yan_yemek, etc.)
    </ResponseField>
    
    <ResponseField name="dates" type="DateInfo[]">
      Array of dates when this meal appears on the menu
      
      <Expandable title="DateInfo properties">
        <ResponseField name="date" type="string">
          ISO date format `YYYY-MM-DD`
        </ResponseField>
        
        <ResponseField name="dayName" type="string">
          Turkish day name
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

### Implementation details

The search endpoint:

1. Scans all `YYYY-MM` directories in `public/data/`
2. Reads the most recent menu file from each month
3. Aggregates unique meals across all dates
4. Deduplicates meals by name (case-insensitive)
5. Returns dates sorted by most recent first

### Caching

Both endpoints use Next.js ISR with a revalidation period of **1 hour (3600 seconds)**.

<Info>
  Menu data is cached and automatically revalidated every hour to balance freshness with performance.
</Info>

### Data source

Menu data is scraped from the official Çukurova University cafeteria website and stored as JSON files in:

```
public/data/YYYY-MM/menu-YYYYMMDD.json
```

### TypeScript usage

```typescript
import type { DayMenu, Meal, MealCategory } from '@/lib/types';

// Fetch menu for a date
async function getMenuForDate(date: string): Promise<DayMenu | null> {
  const response = await fetch(`/api/menu/date/${date}`);
  const data = await response.json();
  
  if (!data.found) {
    return null;
  }
  
  return data.day;
}

// Search all meals
async function searchAllMeals() {
  const response = await fetch('/api/menu/search');
  const data = await response.json();
  return data.meals;
}
```
